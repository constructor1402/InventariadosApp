package com.example.inventariadosapp.admin.obra

import androidx.compose.foundation.background
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.rememberScrollState
import androidx.compose.foundation.verticalScroll
import androidx.compose.material3.*
import androidx.compose.runtime.Composable
import androidx.compose.runtime.LaunchedEffect
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.unit.dp
import androidx.lifecycle.viewmodel.compose.viewModel
import com.example.inventariadosapp.admin.obra.components.ObraButtonRow
import com.example.inventariadosapp.admin.obra.components.ObraTextField

// Esta anotación resuelve los errores de "This material API is experimental..."
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun GestionObraScreen(
    // Inyectamos el ViewModel
    viewModel: ObraViewModel = viewModel()
) {
    // Definimos el color de fondo general de tus diseños (azul/morado muy claro: F3F4F6)
    val screenBackground = Color(0xFFF3F4F6)

    // El scrollState es necesario para que el contenido sea desplazable en pantallas pequeñas
    val scrollState = rememberScrollState()

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("Gestión de Obras") },
                colors = TopAppBarDefaults.topAppBarColors(
                    containerColor = Color(0xFF7986CB), // Color de la barra superior
                    titleContentColor = Color.White
                )
            )
        },
        containerColor = screenBackground
    ) { paddingValues ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(paddingValues)
                .padding(16.dp)
                .background(screenBackground)
                .verticalScroll(scrollState), // Habilitar desplazamiento
            horizontalAlignment = Alignment.CenterHorizontally
        ) {

            // Título principal
            Text(
                text = "Gestión de Obras",
                style = MaterialTheme.typography.headlineMedium,
                modifier = Modifier.padding(bottom = 16.dp)
            )

            // 1. Campo Nombre de Obra
            ObraTextField(
                value = viewModel.nombreObra,
                onValueChange = viewModel::updateNombreObra,
                label = "Nombre de Obra",
                isError = viewModel.nombreObraError,
                errorMessage = if (viewModel.nombreObraError) "Escribe el nombre de la obra." else null
            )

            // 2. Campo Ubicación
            ObraTextField(
                value = viewModel.ubicacion,
                onValueChange = viewModel::updateUbicacion,
                label = "Ubicación",
                isError = viewModel.ubicacionError,
                errorMessage = if (viewModel.ubicacionError) "Ingresa la ubicación." else null
            )

            // 3. Campo Cliente (Debe ser readOnly, su valor se carga desde una búsqueda)
            ObraTextField(
                value = viewModel.clienteNombre.ifBlank { "Cliente asociado" },
                onValueChange = {}, // No editable por el usuario
                label = "Cliente",
                isError = false,
                readOnly = true
            )

            Spacer(Modifier.height(32.dp))

            // 4. Botones de Acción (Se eliminan las advertencias de 'never used' aquí)
            ObraButtonRow(
                onGuardarClick = viewModel::guardarObra,
                onBuscarClick = {
                    // Simulación de búsqueda
                    viewModel.buscarObra("LA PALMA")
                },
                onEliminarClick = viewModel::eliminarObra
            )

            Spacer(Modifier.height(24.dp))

            // 5. Mensaje de Estado (Éxito o Error)
            viewModel.mensajeStatus?.let { status ->
                Card(
                    colors = CardDefaults.cardColors(
                        // Determinar color de fondo basado en el mensaje
                        containerColor = if (status.startsWith("✅")) Color(0xFFE8F5E9) else Color(0xFFFFEBEE)
                    ),
                    modifier = Modifier.fillMaxWidth().padding(horizontal = 8.dp)
                ) {
                    Text(
                        text = status,
                        color = if (status.startsWith("✅")) Color(0xFF388E3C) else Color(0xFFD32F2F),
                        modifier = Modifier.padding(12.dp)
                    )
                }
            }
        }
    }
}
